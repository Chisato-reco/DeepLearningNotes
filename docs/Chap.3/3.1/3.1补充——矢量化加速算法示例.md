# 矢量化加速算法示例
示例代码：
```python
# 导入必要的库
# %matplotlib inline  # 如果在Jupyter Notebook中运行，取消注释以嵌入图表
import math
import time
import numpy as np
import torch
from d2l import torch as d2l  # 使用d2l库中的工具函数[citation:1][citation:3]

# ============================================================
# 1. 定义计时器类（用于基准测试）
# ============================================================
class Timer:  # @save
    """记录多次运行时间的工具类"""
    def __init__(self):
        self.times = []   # 存储每次计时结果
        self.start()      # 初始化时自动启动计时

    def start(self):
        """启动计时器，记录当前时刻"""
        self.tik = time.time()  # 获取当前时间戳（秒）

    def stop(self):
        """停止计时，记录本次运行时长，并返回"""
        self.times.append(time.time() - self.tik)  # 计算时间差并存入列表
        return self.times[-1]  # 返回最近一次记录的时间

    def avg(self):
        """返回所有计时结果的平均值"""
        return sum(self.times) / len(self.times)

    def sum(self):
        """返回所有计时结果的总和"""
        return sum(self.times)

    def cumsum(self):
        """返回累计时间列表（可用于绘制学习曲线等）"""
        return np.array(self.times).cumsum().tolist()

# ============================================================
# 2. 准备测试数据
# ============================================================
n = 10000  # 设置向量长度，即元素数量

# 创建两个全为1的一维张量，形状均为 (n,)
# torch.ones() 创建全1张量，[n] 表示一维张量长度为n
a = torch.ones([n])  # 等价于 torch.ones(n)
b = torch.ones([n])

print(f"向量长度: {len(a)}")
print(f"a 的前10个元素: {a[:10]}")  # 查看前10个元素验证

# ============================================================
# 3. 方法一：使用 Python for 循环（逐元素相加）—— 低效方式
# ============================================================
# 创建全零张量 c，用于存储逐元素相加的结果
c = torch.zeros(n)

# 实例化计时器
timer = Timer()

# 使用 for 循环遍历每个索引，逐元素执行加法
# 这种方式在Python层面进行循环，每次迭代都涉及Python解释器开销
for i in range(n):
    c[i] = a[i] + b[i]  # 通过索引访问并赋值

# 停止计时并打印耗时
time_for = timer.stop()
print(f"for循环耗时: {time_for:.5f} 秒")
# 典型输出: 0.09661 秒 或 1.73559 秒（取决于机器性能）[citation:1][citation:5]

# ============================================================
# 4. 方法二：使用向量化运算（重载的 + 运算符）—— 高效方式
# ============================================================
# 重新启动计时器（会重置时间记录）
timer.start()

# 使用 PyTorch 重载的加法运算符，直接执行向量加法
# 底层调用高度优化的 C++/CUDA 代码，利用SIMD指令集并行计算
d = a + b

# 停止计时并打印耗时
time_vec = timer.stop()
print(f"向量化运算耗时: {time_vec:.5f} 秒")
# 典型输出: 0.00021 秒 或 0.00100 秒（比for循环快几百到上千倍）[citation:1][citation:5]

# ============================================================
# 5. 性能对比分析
# ============================================================
print("\n=== 性能对比 ===")
print(f"for循环耗时: {time_for:.5f} 秒")
print(f"向量化耗时 : {time_vec:.5f} 秒")
print(f"加速比    : {time_for / time_vec:.2f}x")
# 矢量化通常会带来数量级的性能提升[citation:1][citation:3]

# 验证两种方法结果是否一致
print(f"\n结果是否一致: {torch.allclose(c, d)}")  # 应返回 True
```
